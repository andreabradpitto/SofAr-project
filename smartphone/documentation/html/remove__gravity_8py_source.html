57 def anglesCompensate(angles):
58  """!
59  Function used to filter unwanted minimal incoming data fluctuations,
60  due to noise as well as human operator shake
61  @param orientation with respect to X, Y, Z axes
62  @returns returns a filtered version (if necessary) of the input angles
63  """
64  dx = 0.0174 # min angle perceived [rad], about 1 [deg]
65  # reduce sensibility of sensor: minimum precision is dx
66  compensatedAngles = [0, 0, 0]
67 
68  for i in range(0, 3): # i = 0, 1, 2
69  if abs(angles[i] / dx) >= 1:
70  compensatedAngles[i] = angles[i]
71 
72  return compensatedAngles

75 def removeGravity(lin_acc, Rot_m):
76  """!
77  This function is the one that actually carries out the gravity removal
78  @param lin_acc linear acceleration data incoming from the accelerometer
79  @param Rot_m rotation matrix computed with eulerAnglesToRotationMatrix()
80  @returns the output is the same linear acceleration vector provided, but without the influence of the gravity
81  """
82  g = [0, 0, 9.81]
83 
84  # rotate g vector in the current frame
85  g_frame_i = np.dot(Rot_m, g)
86  g = [0, 0, 9.81]
87  g_removed = [0, 0, 0] # define linear acceleration without gravity
88 
89  for i in range(0, 3):
90  if lin_acc[i] >= 0:
91  g_removed[i] = lin_acc[i] - abs(g_frame_i[i])
92  if lin_acc[i] < 0:
93  g_removed[i] = lin_acc[i] + abs(g_frame_i[i])
94 
95  return g_removed

110 def callback(data):
111  """!
112  This is the callback function: it is invoked every time there is incoming data and
113  has the duty of calling all the previously mentioned functions as well as eulerAnglesToRotationMatrix().
114  After these calls, it finally invokes the talker(). In order to achieve all of this,
115  it translates incoming quaternions into euler angles beforehand
116  @param data incoming from the imu sensor
117  """
118  global index, fileSet
119 
120  # get data
121  orientation = [data.orientation.x, data.orientation.y,
122  data.orientation.z, data.orientation.w]
123 
124  # transform quaternion to euler angles
125  tempAngles = tf.transformations.euler_from_quaternion(orientation, "sxyz")
126 
127  angular_velocity = [data.angular_velocity.x,
128  data.angular_velocity.y, data.angular_velocity.z]
129  linear_acceleration = [data.linear_acceleration.x,
130  data.linear_acceleration.y, data.linear_acceleration.z]
131 
132  angles = anglesCompensate(tempAngles)
133 
134  rot_matrix = eulerAnglesToRotationMatrix(angles)
135 
136  lin_acc_no_g = removeGravity(linear_acceleration, rot_matrix)
137 
138  if fileSet == 0:
139  dataFileInitializer()
140 
141  if flagWriteData == 1:
142  # store data into .csv files in order to analyse them offline
143  storeDataInFiles(abs_file_path1, 'a', lin_acc_no_g)
144 
145  angles_in_deg = [(angles[0]*180) / math.pi,
146  (angles[1]*180) / math.pi, (angles[2]*180) / math.pi]
147  storeDataInFiles(abs_file_path2, 'a', angles_in_deg)
148  storeDataInFiles(abs_file_path3, 'a', angular_velocity)
149 
150  index += 1 # update index
151 
152  # modify Imu message to be sent
153  data.linear_acceleration.x = lin_acc_no_g[0]
154  data.linear_acceleration.y = lin_acc_no_g[1]
155  data.linear_acceleration.z = lin_acc_no_g[2]
156 
157  # publish message
158  try:
159  talker(data)
160  except rospy.ROSInterruptException:
161  pass
