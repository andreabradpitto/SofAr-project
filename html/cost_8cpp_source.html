46 void computeCostResponse(VectorXd q,VectorXd qdot1,VectorXd qdot2,MatrixXd Q2,math_pkg::Cost::Response &res) {
47  VectorXd qdot1opt1,qdot1opt2,qdot2opt1,qdot2opt2; // initialization
48 
49  double cond1,cond2;
50 
51  // Each optimized velocity vector is given by: non-optimized vector + G * z, where z is a NJOINTSx1 vector.
52  // Optimization n. 1: minimize qdot
53  /* qdot1opt1 and qdot2opt1 are computed according to the paper "A Novel Practical Technique to Integrate Inequality Control
54  * Objectives and Task Transitions in Priority Based Control" by Casalino & Simetti, p. 20, sec. 4.4.*/
55  MatrixXd IdMinusQ2 = ID_MATRIX_NJ - Q2;
56  MatrixXd toPinv = Q2.transpose() * Q2 + 0.1 * IdMinusQ2.transpose()*(IdMinusQ2);
57  MatrixXd temp1 = -regPinv(toPinv,ID_MATRIX_NJ,ID_MATRIX_NJ,ETA,cond1) * Q2.transpose();
58  qdot1opt1 = qdot1 + Q2*temp1*qdot1; // optimize solution 1
59  qdot2opt1 = qdot2 + Q2*temp1*qdot2; // optimize solution 2
60 
61  // Optimization n. 2: stay close to favourite pose (initial pose)
62  MatrixXd Q2Sharp = regPinv(Q2,ID_MATRIX_NJ,ID_MATRIX_NJ,ETA,cond2);
63  VectorXd qdot_fav = 0.001 * q;
64  int multFact = 0;
65  if (cond2 < 100) multFact = 1;
66  qdot1opt2 = qdot1 + multFact*Q2*Q2Sharp*(qdot_fav - qdot1); // optimize solution 1
67  qdot2opt2 = qdot2 + multFact*Q2*Q2Sharp*(qdot_fav - qdot2); // optimize solution 2
68 
69  // Fill the response object.
70  res.qdot1opt1.velocity = vector<double> (qdot1opt1.data(),qdot1opt1.data()+qdot1opt1.size());
71  res.qdot1opt2.velocity = vector<double> (qdot1opt2.data(),qdot1opt2.data()+qdot1opt2.size());
72  res.qdot2opt1.velocity = vector<double> (qdot2opt1.data(),qdot2opt1.data()+qdot2opt1.size());
73  res.qdot2opt2.velocity = vector<double> (qdot2opt2.data(),qdot2opt2.data()+qdot2opt2.size());
74 }
