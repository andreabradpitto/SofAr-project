78 void DoCalib::imuCallback(sensor_msgs::Imu::ConstPtr imu)
79 {
80  bool accepted;
81 
82  switch (state_)
83  {
84  case START:
85  calib_.beginCalib(6*measurements_per_orientation_, reference_acceleration_);
86  state_ = SWITCHING;
87  break;
88 
89 
90  case SWITCHING:
91  if (orientations_.empty())
92  {
93  state_ = COMPUTING;
94  }
95  else
96  {
97  current_orientation_ = orientations_.front();
98 
99  orientations_.pop();
100  measurements_received_ = 0;
101 
102  std::cout << "Orient IMU with " << orientation_labels_[current_orientation_] << " axis up and press Enter";
103  std::cin.get();
104  std::cout << "Recording measurements...";
105 
106  state_ = RECEIVING;
107  }
108  break;
109 
110 
111  case RECEIVING:
112  accepted = calib_.addMeasurement(current_orientation_,
113  imu->linear_acceleration.x,
114  imu->linear_acceleration.y,
115  imu->linear_acceleration.z);
116 
117  measurements_received_ += accepted ? 1 : 0;
118  if (measurements_received_ >= measurements_per_orientation_)
119  {
120  std::cout << " Done." << std::endl;
121  state_ = SWITCHING;
122  }
123  break;
124 
125 
126  case COMPUTING:
127  std::cout << "Computing calibration parameters...";
128  if (calib_.computeCalib())
129  {
130  std::cout << " Success!" << std::endl;
131 
132  std::cout << "Saving calibration file...";
133  if (calib_.saveCalib(output_file_))
134  {
135  std::cout << " Success!" << std::endl;
136  }
137  else
138  {
139  std::cout << " Failed." << std::endl;
140  }
141  }
142  else
143  {
144  std::cout << " Failed.";
145  ROS_ERROR("Calibration failed");
146  }
147  state_ = DONE;
148  break;
149 
150 
151  case DONE:
152  break;
153  }
154 }
