47 ApplyCalib::ApplyCalib() :
48  gyro_sample_count_(0),
49  gyro_bias_x_(0.0),
50  gyro_bias_y_(0.0),
51  gyro_bias_z_(0.0)
52 {
53  ros::NodeHandle nh;
54  ros::NodeHandle nh_private("~");
55 
56  std::string calib_file;
57  nh_private.param<std::string>("calib_file", calib_file, "imu_calib.yaml");
58 
59  if (!calib_.loadCalib(calib_file) || !calib_.calibReady())
60  {
61  ROS_FATAL("Calibration could not be loaded");
62  ros::shutdown();
63  }
64 
65  nh_private.param<bool>("calibrate_gyros", calibrate_gyros_, true);
66  nh_private.param<int>("gyro_calib_samples", gyro_calib_samples_, 100);
67 
68  int queue_size;
69  nh_private.param<int>("queue_size", queue_size, 5);
70 
71  raw_sub_ = nh.subscribe("raw", queue_size, &ApplyCalib::rawImuCallback, this);
72  corrected_pub_ = nh.advertise<sensor_msgs::Imu>("corrected", queue_size);
73 }

75 void ApplyCalib::rawImuCallback(sensor_msgs::Imu::ConstPtr raw)
76 {
77  if (calibrate_gyros_)
78  {
79  ROS_INFO_ONCE("Calibrating gyros; do not move the IMU");
80 
81  // recursively compute mean gyro measurements
82  gyro_sample_count_++;
83  gyro_bias_x_ = ((gyro_sample_count_ - 1) * gyro_bias_x_ + raw->angular_velocity.x) / gyro_sample_count_;
84  gyro_bias_y_ = ((gyro_sample_count_ - 1) * gyro_bias_y_ + raw->angular_velocity.y) / gyro_sample_count_;
85  gyro_bias_z_ = ((gyro_sample_count_ - 1) * gyro_bias_z_ + raw->angular_velocity.z) / gyro_sample_count_;
86 
87  if (gyro_sample_count_ >= gyro_calib_samples_)
88  {
89  ROS_INFO("Gyro calibration complete! (bias = [%.3f, %.3f, %.3f])", gyro_bias_x_, gyro_bias_y_, gyro_bias_z_);
90  calibrate_gyros_ = false;
91  }
92 
93  return;
94  }
95 
96  sensor_msgs::Imu corrected = *raw;
97 
98  calib_.applyCalib(raw->linear_acceleration.x, raw->linear_acceleration.y, raw->linear_acceleration.z,
99  &corrected.linear_acceleration.x, &corrected.linear_acceleration.y, &corrected.linear_acceleration.z);
100 
101  corrected.angular_velocity.x -= gyro_bias_x_;
102  corrected.angular_velocity.y -= gyro_bias_y_;
103  corrected.angular_velocity.z -= gyro_bias_z_;
104 
105  corrected_pub_.publish(corrected);
106 }
